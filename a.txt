create table yzz_book_borrow(
	id int(5) auto_increment primary key,
	userid char(15) not null,
	bookid int(5) not null,
	borrowtime int(15) not null,
	returntime int(15),
	status int(3) not null default 0,
	xjcs int(1) not null default 0,
	foreign key(userid) references yzz_user(id),
	foreign key(bookid) references yzz_book(id)
)
create table yzz_book_booked(
	id int(5) auto_increment primary key,
	userid char(15) not null,
	bookid int(5) not null,
	bookedtime int(15) not null,
	status int(2) not null default 0,
	foreign key(userid) references yzz_user(id),
	foreign key(bookid) references yzz_book(id)
)
yzz_book_booked表单中
如果为0表示还有效
status如果为1则表示该预约失效


0 未还
1 已还
2 逾期
3 被预约
4 续借未还
